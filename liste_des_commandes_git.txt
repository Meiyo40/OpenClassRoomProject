git push  //Upload le dernier commit, doit préciser la remote (origin par convention pour github) et la branche de dev en argument ex: git push origin master
git pull //Permet de télécharger la dernière mise à jour(dernier commit) du repertoire distant, utilisation des arguments similaire a la commande push, ex git pull origin master
git add // permet d'ajouter un nouveau fichier/rep à l'index pour l'ajouter au prochain commit
git clone //Permet de télécharger l'intrégralité d'un repertoire git en utilisant l'adresse de ce dernier disponible sur github par exemple
git commit -am/m "Text" // Permet de valider un commit suite à un ajout avant de l'uploader en laissant un message (argument -m), possibilité d'ajout avec (-a)
git status //vérifie si les fichiers locaux sont à jour (et ajouté à l'index) où si un commit est nécessaire
git log //Donne une liste des commits courant et précédant sur le répertoire en cours
ls //Commande bash liste les fichiers présent dans le repertoire courant
mkdir // Commande bash permet la création d'un repertoire du fichier courant ou ailleurs par un chemin relatif ou absolu
cd //Permet de se positionner dans les différents fichier en utilisant leurs localisation exemple C:\dev\openclassroom
